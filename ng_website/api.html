Java 
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

public class NextGPUApiExample {
    public static void main(String[] args) {
        // API端点
        String url = "https://api.nextgpu.com/v1/images/generations";
        
        // 请求体JSON
        String jsonBody = "{" +
            "\"prompt\": \"A beautiful landscape with mountains and a lake, digital art\"," +
            "\"model\": \"star-3-alpha\"," +
            "\"size\": \"1024x1024\"," +
            "\"steps\": 20," +
            "\"guidance_scale\": 7.5," +
            "\"num_images\": 1" +
        "}";
        
        // 创建HTTP请求
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(url))
            .header("Content-Type", "application/json")
            .header("Authorization", "Bearer YOUR_API_KEY")
            .POST(HttpRequest.BodyPublishers.ofString(jsonBody))
            .build();
        
        // 发送请求并处理响应
        HttpClient client = HttpClient.newHttpClient();
        client.sendAsync(request, HttpResponse.BodyHandlers.ofString())
            .thenApply(HttpResponse::body)
            .thenAccept(response -> {
                // 解析响应JSON获取图像URL
                // 这里简化处理，实际应用中应使用JSON解析库
                System.out.println("图像生成成功! 响应内容:" + response);
            })
            .exceptionally(e -> {
                System.err.println("请求失败: " + e.getMessage());
                return null;
            })
            .join();
    }
}
PHP 
<?php
// API端点
$url = 'https://api.nextgpu.com/v1/images/generations';

// 请求头
$headers = [
    'Content-Type: application/json',
    'Authorization: Bearer YOUR_API_KEY'
];

// 请求体
$data = [
    'prompt' => 'A beautiful landscape with mountains and a lake, digital art',
    'model' => 'star-3-alpha',
    'size' => '1024x1024',
    'steps' => 20,
    'guidance_scale' => 7.5,
    'num_images' => 1
];

// 初始化cURL
$ch = curl_init($url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));

// 发送请求
$response = curl_exec($ch);
$httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);

if (curl_errno($ch)) {
    // 处理cURL错误
    echo '请求失败: ' . curl_error($ch);
} else {
    if ($httpCode == 200) {
        // 解析响应
        $result = json_decode($response, true);
        $imageUrl = $result['data'][0]['url'];
        echo "图像生成成功! 访问URL: $imageUrl";
    } else {
        echo "请求失败，HTTP状态码: $httpCode";
        echo "错误信息: $response";
    }
}

// 关闭cURL资源
curl_close($ch);
?>
